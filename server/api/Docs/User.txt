
Each req call api => Check res.msg. If 'success' => OK else FAIL

// Signup new account
Method: POST
Auth: No
Req.body:
{
	name: String
	email: String
	password: String
}
URL: http://localhost:8080/users/signup
If 'success' => Check email => Verify

Auth: No

Auth: No

{
	email: String
}
